02 - AJAX + XML: Load Sản Phẩm từ File XML

Lý thuyết: AJAX là gì?

AJAX (Asynchronous JavaScript And XML): Kỹ thuật giúp trang web gửi/nhận dữ liệu từ server mà không cần tải lại trang.

Dùng đối tượng XMLHttpRequest (XHR) tích hợp sẵn trong trình duyệt.

Các bước thực hiện AJAX (Luồng hoạt động)

1. Tạo đối tượng XMLHttpRequest → let xhr = new XMLHttpRequest();
2. Cấu hình request → xhr.open("GET", "url", true/false);
3. Đăng ký hàm xử lý response → xhr.onreadystatechange = function() { ... }
4. Gửi request → xhr.send();
5. Trong hàm callback: kiểm tra readyState == 4 && status == 200 → xử lý dữ liệu

Bảng thuộc tính XMLHttpRequest quan trọng

Thuộc tínhÝ nghĩa
onreadystatechangeHàm callback gọi khi readyState thay đổi
readyState0: chưa khởi tạo | 1: đang kết nối | 2: đã nhận request | 3: đang xử lý | 4: hoàn thành
status200: OK | 403: Forbidden | 404: Not Found
responseTextTrả về dữ liệu dạng chuỗi text
responseXMLTrả về dữ liệu dạng XML Document (có thể dùng DOM)

Code mẫu AJAX - Template đầy đủ

// ====== TEMPLATE AJAX ĐỌC FILE XML ======

function loadDoc() {
  // BƯỚC 1: Tạo đối tượng XMLHttpRequest
  var xhr = new XMLHttpRequest();

  // BƯỚC 2: Cấu hình request
  // open(method, url, async)
  // method: "GET" hoặc "POST"
  // url: đường dẫn đến file/API
  // async: true (bất đồng bộ) | false (đồng bộ - dùng trong slide)
  xhr.open("GET", "./data/products.xml", false);

  // BƯỚC 3: Đăng ký hàm xử lý khi có response
  xhr.onreadystatechange = function() {
    if (xhr.readyState == 4 && xhr.status == 200) {
      // Thành công → xử lý dữ liệu
      updatePage(this); // this = đối tượng xhr
    } else {
      // Thất bại
      alert("Không lấy được dữ liệu");
    }
  };

  // BƯỚC 4: Gửi request
  xhr.send();
}

// ====== HÀM XỬ LÝ DỮ LIỆU XML ======
function updatePage(xhr) {
  // Lấy XML Document từ response
  var xmlDoc = xhr.responseXML;

  // Lấy tất cả phần tử <product>
  var products = xmlDoc.getElementsByTagName("product");

  var itemLayout = "";
  for (var i = 0; i < products.length; i++) {
    // Lấy giá trị từng field: tên_thẻ → [0] → childNodes[0] → nodeValue
    var id = products[i].getElementsByTagName("id")[0].childNodes[0].nodeValue;
    var name = products[i].getElementsByTagName("name")[0].childNodes[0].nodeValue;
    var price = products[i].getElementsByTagName("price")[0].childNodes[0].nodeValue;

    itemLayout += '<div class="wrap__item">' + id + '<br><b>' +
                  name + '</b><br>' + price + '</div>';
  }
  document.getElementById("content").innerHTML = itemLayout;
}
