04 - JSON Cơ Bản: Object & Array

1. JSON là gì?

JSON (JavaScript Object Notation): Định dạng dữ liệu phổ biến dạng key:value.
Dùng để trao đổi dữ liệu giữa server và client trên web.
Nhẹ hơn XML, dễ đọc hơn, được hỗ trợ trực tiếp trong JavaScript.

2. JSON Object vs JSON Array

JSON Object — dùng { }
{
  "id": 1,
  "name": "Heineken",
  "price": 19000
}
Truy cập: obj.name hoặc obj["name"]
JSON Array — dùng [ ]
[
  { "id": 1, "name": "Heineken" },
  { "id": 2, "name": "Tiger" },
  { "id": 3, "name": "Sapporo" }
]
Truy cập: arr[0].name

3. Các hàm JSON quan trọng

HàmÝ nghĩaVí dụ
JSON.parse(string) Chuyển chuỗi JSON → Object JS JSON.parse('{"name":"A"}') → Object
JSON.stringify(obj) Chuyển Object JS → chuỗi JSON JSON.stringify({name:"A"}) → String

4. Code mẫu: Đọc JSON Array (dữ liệu inline)

// ====== TEMPLATE ĐỌC JSON ARRAY INLINE ======

function loadJson() {
  let htmlString = '';

  // Dữ liệu JSON Array khai báo trực tiếp trong JS
  var products = [
    { "id": 1, "name": "Heineken", "price": 19000 },
    { "id": 2, "name": "Tiger", "price": 18000 },
    { "id": 3, "name": "Sapporo", "price": 21000 }
  ];

  // Duyệt mảng bằng for...of (cú pháp mới)
  for (let p of products) {
    htmlString += '<div class="item-card">' +
      p.id + '<br><b>' + p.name + '</b><br>' + p.price +
      '</div>';
  }

  // Gán vào DOM
  document.getElementById("content").innerHTML = htmlString;
}

5. Code mẫu: JSON.parse() - Chuyển chuỗi thành Object

// Kịch bản: nhận dữ liệu từ server dưới dạng chuỗi, cần parse thành object

var jsonString = '{"id":1,"name":"Heineken","price":19000}'; // chuỗi JSON
var obj = JSON.parse(jsonString); // parse → object

console.log(obj.name); // → "Heineken"
console.log(obj.price); // → 19000

// Ngược lại: object → chuỗi
var str = JSON.stringify(obj); // → '{"id":1,"name":"Heineken","price":19000}'

6. Code mẫu: JSON Object lồng nhau (nested)

// JSON có thể lồng nhau: object chứa array, array chứa object
var data = {
  "store": "Bia Shop",
  "products": [ // property "products" là một Array
    { "id": 1, "name": "Heineken" },
    { "id": 2, "name": "Tiger" }
  ]
};

console.log(data.store); // → "Bia Shop"
console.log(data.products[0].name); // → "Heineken"
console.log(data.products.length); // → 2
