2. Code mẫu: Fetch từ API → hiển thị danh sách users

// ====== TEMPLATE: FETCH JSON TỪ API ONLINE ======

function loadJson() {
  var myList = document.querySelector('ul');

  // URL của API (thay bằng API thật của bạn)
  fetch("https://jsonplaceholder.typicode.com/users")
    .then(function(response) {
      if (!response.ok) {
        throw new Error("HTTP error, status = " + response.status);
      }
      return response.json(); // API trả về Array trực tiếp (không bọc trong object)
    })
    .then(function(users) { // users là Array các user object
      for (let user of users) {
        var listItem = document.createElement('li');
        listItem.innerHTML = '<strong>' + user.id + '</strong>';
        listItem.innerHTML += user.name;
        listItem.innerHTML += ' >> <strong>' + user.email + '</strong>';
        myList.appendChild(listItem);
      }
    })
    .catch(function(error) {
      alert(error.message);
    });
}
