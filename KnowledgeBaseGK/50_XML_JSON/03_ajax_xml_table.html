Code mẫu: AJAX load XML → hiển thị Table

// ====== LOAD FILE XML → HIỂN THỊ DẠNG BẢNG ======

function loadDoc() {
  var xhr = new XMLHttpRequest();
  xhr.open("GET", "./data/cd_catalog.xml", false);
  xhr.onreadystatechange = function() {
    if (xhr.readyState == 4 && xhr.status == 200) {
      updatePage(this);
    } else {
      alert("Không lấy được dữ liệu");
    }
  };
  xhr.send();
}

function updatePage(xhr) {
  var xmlDoc = xhr.responseXML; // Lấy XML Document

  // Khởi tạo bảng với header row
  var table = "<tr><th>Artist</th><th>Title</th></tr>";

  // Lấy tất cả phần tử <cd>
  var cds = xmlDoc.getElementsByTagName("cd");

  for (var i = 0; i < cds.length; i++) {
    table += "<tr><td>" +
      cds[i].getElementsByTagName("artist")[0].childNodes[0].nodeValue +
      "</td><td>" +
      cds[i].getElementsByTagName("title")[0].childNodes[0].nodeValue +
      "</td></tr>";
  }

  // Gán vào element có id="content"
  document.getElementById("content").innerHTML = table;
}
