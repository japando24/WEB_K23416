⚡ 03 — Điều Kiện & Vòng Lặp

1. if / else if / else

// Cú pháp cơ bản if (điều_kiện) { // chạy nếu đúng } else if (điều_kiện_2) { // chạy nếu điều_kiện_1 sai, điều_kiện_2 đúng } else { // chạy nếu tất cả sai } // Ví dụ thực tế: xếp loại học lực let score = 85; if (score >= 90) { console.log("Xuất sắc"); } else if (score >= 80) { console.log("Giỏi"); // ← sẽ in cái này } else if (score >= 70) { console.log("Khá"); } else if (score >= 50) { console.log("Trung bình"); } else { console.log("Yếu"); } // if ngắn — không cần {} nếu chỉ 1 câu lệnh (không khuyến khích) if (score > 50) console.log("Đậu"); // Điều kiện phức hợp let age = 20, hasID = true; if (age >= 18 && hasID) { console.log("Được vào"); }

2. switch / case

// switch so sánh === (strict) let day = 3; switch (day) { case 1: console.log("Thứ Hai"); break; // ⚠️ PHẢI có break, không thì "fall-through"! case 2: console.log("Thứ Ba"); break; case 3: console.log("Thứ Tư"); // ← in cái này break; case 6: case 7: // 2 case chung 1 block console.log("Cuối tuần!"); break; default: // không bắt buộc, chạy khi không khớp case nào console.log("Không xác định"); } // ⚠️ Thiếu break → fall-through (chạy cả case tiếp theo!) switch (2) { case 1: console.log("1"); // bỏ qua case 2: console.log("2"); // in case 3: console.log("3"); // IN LUÔN vì không có break! default: console.log("d");// IN LUÔN! } // Kết quả: "2", "3", "d" — đây là bug phổ biến!

3. Vòng Lặp for

// Cú pháp: for (khởi tạo; điều kiện; cập nhật) for (let i = 0; i < 5; i++) { console.log(i); // 0, 1, 2, 3, 4 } // Đếm ngược for (let i = 5; i > 0; i--) { console.log(i); // 5, 4, 3, 2, 1 } // Duyệt mảng const fruits = ["Táo", "Chuối", "Cam"]; for (let i = 0; i < fruits.length; i++) { console.log(i + ": " + fruits[i]); } // break — thoát vòng lặp for (let i = 0; i < 10; i++) { if (i === 5) break; // dừng ở i=5 console.log(i); // 0, 1, 2, 3, 4 } // continue — bỏ qua lần lặp hiện tại for (let i = 0; i < 5; i++) { if (i === 2) continue; // bỏ qua i=2 console.log(i); // 0, 1, 3, 4 }

4. for...in và for...of

// for...in — duyệt KEY (tên thuộc tính) của Object const person = { name: "An", age: 20, city: "HCM" }; for (let key in person) { console.log(key + ": " + person[key]); // "name: An", "age: 20", "city: HCM" } // for...in cũng dùng được với mảng (lấy index) const arr = ["a", "b", "c"]; for (let i in arr) { console.log(i, arr[i]); // "0 a", "1 b", "2 c" } // for...of — duyệt VALUE của iterable (Array, String, Map, Set) const colors = ["Đỏ", "Xanh", "Vàng"]; for (let color of colors) { console.log(color); // "Đỏ", "Xanh", "Vàng" } // for...of với String for (let char of "Hello") { console.log(char); // H, e, l, l, o } // So sánh for...in vs for...of với mảng const nums = [10, 20, 30]; for (let i in nums) console.log(i); // "0", "1", "2" (index) for (let n of nums) console.log(n); // 10, 20, 30 (value) nums.forEach((n, i) => console.log(i, n)); // 0 10, 1 20, 2 30
📌 Nhớ nhanh:
• for...in → lấy KEY/INDEX (dùng cho Object)
• for...of → lấy VALUE (dùng cho Array/String)

5. while và do...while

// while — kiểm tra điều kiện TRƯỚC, có thể không chạy lần nào let i = 0; while (i < 5) { console.log(i); // 0, 1, 2, 3, 4 i++; // ⚠️ Phải cập nhật biến, không sẽ lặp vô tận! } // Vòng lặp đọc input (ví dụ điển hình) let password = ""; while (password !== "1234") { password = prompt("Nhập mật khẩu:"); // lặp đến khi đúng } console.log("Đúng mật khẩu!"); // do...while — chạy ÍT NHẤT 1 LẦN, kiểm tra sau let count = 0; do { console.log("Lần " + count); count++; } while (count < 3); // "Lần 0", "Lần 1", "Lần 2" // Ví dụ: khi điều kiện sai ngay từ đầu let x = 10; do { console.log("Chạy 1 lần!"); // in dù điều kiện sai! } while (x < 5); while (x < 5) { console.log("Không bao giờ in!"); // không in }

6. So Sánh Các Vòng Lặp

Vòng lặpDùng khi nàoĐặc điểm
for (i)Biết trước số lần, cần indexKiểm soát tốt nhất
for...inDuyệt key/property của ObjectTrả về key (string)
for...ofDuyệt value của Array/String/iterableTrả về value trực tiếp
whileKhông biết trước số lầnCó thể không chạy
do...whileCần chạy ít nhất 1 lầnLuôn chạy ít nhất 1 lần
forEachDuyệt mảng (functional style)Không dùng được break

📌 Tóm Tắt Quan Trọng

  • switch/case phải có break, không thì "fall-through" sang case tiếp theo
  • for...in → KEY (index/tên thuộc tính) | for...of → VALUE
  • while: kiểm tra trước — có thể không chạy lần nào
  • do...while: kiểm tra sau — luôn chạy ít nhất 1 lần
  • break: thoát vòng lặp | continue: bỏ qua lần lặp hiện tại
  • switch dùng === nên case "3": khác case 3:
